import Foundation

func solution(_ str_list:[String]) -> [String] {
    var getLR = str_list.enumerated().filter{$0.element == "l" || $0.element == "r"}.map{$0.offset}
    
    for i in getLR {
        if str_list[i] == "l" {
            return Array(str_list[...(i-1)])
        } else {
            return Array(str_list[(i+1)...])
        }
    }
    return []
}
```
